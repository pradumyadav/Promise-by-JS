function printNumbersWithDelay(num, delay) {

let counter = 1;

return new Promise((resolve, reject) => {

const interval = setInterval(() => {

console.log(counter);

counter++;

if (counter > num) {

clearInterval(interval);

resolve();

}

}, delay);

});

}

Q7>Create examples to explain promise function.

1. Resolving a Promise

const promise = new Promise((resolve, reject) => {

setTimeout(() => {

resolve("Task completed successfully!");

}, 3000);

});

promise.then((result) => {

console.log(result);

});

2. Rejecting a Promise.

const promise = new Promise((resolve, reject) => {

setTimeout(() => {

reject("Task failed!");

}, 4000);

});

promise.catch((error) => {

console.log(error);

});

3.Chaining Promises.

const promise1 = new Promise((resolve, reject) => {

setTimeout(() => {

resolve("Task 1 completed successfully!");

}, 2000);

});

const promise2 = new Promise((resolve, reject) => {

setTimeout(() => {

resolve("Task 2 completed successfully!");

}, 6000);

});

promise1.then((result) => {

console.log(result);

return promise2;

}).then((result) => {

console.log(result);

});

async function myFunction() {

myFunction();

Q9> Create examples to explain promise.all function.

const promise2 = Promise.resolve(2);
